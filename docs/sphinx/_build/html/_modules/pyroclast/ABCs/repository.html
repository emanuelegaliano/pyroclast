    def get(self, criteria: MapCriteria) -> RasterMap:
        """Return the single map matching criteria. Raises if not exactly one."""
        results = self.matching(criteria)
        if len(results) != 1:
            raise ValueError(
                f"Expected exactly one result for {criteria!r}, got {len(results)}"
            )
        return results[0]
